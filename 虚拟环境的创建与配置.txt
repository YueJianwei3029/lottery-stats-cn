my_data_17500/               # 项目根目录
├── main.py                  # 项目初始化入口（核心！配置数据库连接、启动FastAPI服务）
├── scheme/                  # pydantic模型层：定义请求模型、响应模型
├── model/                   # Model层：定义MySQL表结构（所有表的模型都放在这里）
│   └── __init__.py          # 空文件，标记models为Python包
│   └── super_lottery.py     # super_lottery表模型
│   └──
├── dao/                     # Dao层：数据访问层，专门做数据库增删改查操作
│   └── __init__.py          # 空文件，标记dao为Python包
│   └──
│   └──
├── api/                     # 路由层（子路由）：拆分接口，避免所有接口写在main.py
│   └── __init__.py          # 空文件，标记api为Python包
│   └── super_lottery.py     # super_lottery子路由
│   └──
└── database.py              # 数据库配置文件：存储数据库连接信息、提供会话
